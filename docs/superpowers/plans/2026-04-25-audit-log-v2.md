- [ ] **Step 4: Rewrite `ListAuditLogs` signature + query**

Replace the existing implementation:

```go
func (r *Repository) ListAuditLogs(ctx context.Context,
    serverID int64, actorID *int64, action, target string,
    limit, offset int,
) ([]AuditLog, int, error) {
    args := []any{serverID}
    where := `WHERE sal.server_id = $1`
    idx := 2
    if actorID != nil {
        where += fmt.Sprintf(` AND sal.actor_id = $%d`, idx)
        args = append(args, *actorID)
        idx++
    }
    if action != "" {
        where += fmt.Sprintf(` AND sal.action = $%d`, idx)
        args = append(args, action)
        idx++
    }
    if target != "" {
        where += fmt.Sprintf(` AND sal.target_name ILIKE $%d ESCAPE '\'`, idx)
        args = append(args, "%"+escapeLike(target)+"%")
        idx++
    }

    var total int
    if err := r.db.QueryRowContext(ctx,
        `SELECT COUNT(*) FROM server_audit_logs sal `+where, args...).Scan(&total); err != nil {
        return nil, 0, err
    }

    args = append(args, limit, offset)
    rows, err := r.db.QueryContext(ctx,
        `SELECT sal.id, sal.server_id, sal.actor_id, sal.actor_username,
                COALESCE(actor.avatar_url, ''),
                sal.action,
                COALESCE(sal.target_id, ''), COALESCE(sal.target_type, ''),
                COALESCE(sal.target_name, ''),
                COALESCE(target.avatar_url, ''),
                sal.changes, COALESCE(sal.reason, ''), sal.created_at
         FROM server_audit_logs sal
         LEFT JOIN users actor ON actor.id = sal.actor_id
         LEFT JOIN users target ON sal.target_type = 'user'
                              AND target.id::text = sal.target_id
         `+where+
            fmt.Sprintf(` ORDER BY sal.created_at DESC LIMIT $%d OFFSET $%d`, idx, idx+1),
        args...)
    if err != nil {
        return nil, 0, err
    }
    defer rows.Close()

    var logs []AuditLog
    for rows.Next() {
        var l AuditLog
        if err := rows.Scan(&l.ID, &l.ServerID, &l.ActorID, &l.ActorUsername,
            &l.ActorAvatarURL,
            &l.Action,
            &l.TargetID, &l.TargetType, &l.TargetName,
            &l.TargetAvatarURL,
            &l.Changes, &l.Reason, &l.CreatedAt); err != nil {
            return nil, 0, err
        }
        logs = append(logs, l)
    }
    return logs, total, rows.Err()
}
```

Note `target.id::text = sal.target_id` — `target_id` is TEXT in the schema, and `users.id` is BIGINT, so cast to TEXT for the join. The `sal.target_type = 'user'` predicate gates the join so only user-target rows get matched (avoids accidental matches on numeric channel IDs that happen to equal a user ID).

```ts
export interface AuditLogEntry {
  id: string;
  server_id: string;
  actor_id?: string;
  actor_username: string;
  actor_avatar_url?: string;
  action: string;
  target_id?: string;
  target_type?: string;
  target_name?: string;
  target_avatar_url?: string;
  changes?: Record<string, unknown>;  // loosened — see ChangeDiff overhaul
  reason?: string;
  created_at: string;
}

export async function getAuditLog(
  serverId: string,
  params: { limit?: number; offset?: number; action?: string; actorId?: string; target?: string }
): Promise<{ logs: AuditLogEntry[]; total: number }> {
  const qs = new URLSearchParams();
  if (params.limit !== undefined) qs.set('limit', String(params.limit));
  if (params.offset !== undefined) qs.set('offset', String(params.offset));
  if (params.action) qs.set('action', params.action);
  if (params.actorId) qs.set('actor_id', params.actorId);
  if (params.target) qs.set('target', params.target);
  const qsStr = qs.toString();
  return apiClient.get(`/servers/${serverId}/audit-log${qsStr ? `?${qsStr}` : ''}`);
}
```

- [ ] **Step 7: Overhaul `ChangeDiff`**

Replace it with a renderer that:
1. Detects the shape: `{before, after}` vs flat-keyed (e.g., overwrite entries with `overwrite_target_*` siblings).
2. For each changed key, renders a row.
3. Special-cases:
   - `permissions`, `allow`, `deny`, `granted_permissions` → decode bitfield to permission name list, render added (in `after` not in `before`) with `+` and removed with `-`.
   - `color` → 12×12 swatch + hex code.
   - Booleans → "Yes"/"No".
   - Arrays → comma-joined or "N items".
   - Default → `JSON.stringify`.

Pseudocode:

```tsx
import { PERM_ADMINISTRATOR, PERM_MANAGE_SERVER, /* ... all perm consts ... */ } from '../../lib/permissions';

const PERM_NAMES: Array<[bigint, string]> = [
  [PERM_ADMINISTRATOR, 'Administrator'],
  [PERM_MANAGE_SERVER, 'Manage Server'],
  // ... etc, mirror the constant order
];

function decodePerms(bits: bigint): string[] {
  return PERM_NAMES.filter(([b]) => (bits & b) !== 0n).map(([, name]) => name);
}

function PermDiff({ before, after }: { before: bigint; after: bigint }) {
  const beforeSet = new Set(decodePerms(before));
  const afterSet = new Set(decodePerms(after));
  const added   = [...afterSet].filter(n => !beforeSet.has(n));
  const removed = [...beforeSet].filter(n => !afterSet.has(n));
  if (added.length === 0 && removed.length === 0) return null;
  return (
    <span>
      {added.map(n => <span key={n} className="audit-diff__perm-add">+ {n}</span>)}
      {removed.map(n => <span key={n} className="audit-diff__perm-remove">- {n}</span>)}
    </span>
  );
}

function ColorSwatch({ hex }: { hex: string }) {
  return (
    <span className="audit-diff__color-row">
      <span className="audit-diff__color-swatch" style={{ background: hex }} />
      <code>{hex}</code>
    </span>
  );
}

function renderValue(key: string, value: unknown): React.ReactNode {
  if (key === 'color' && typeof value === 'string') return <ColorSwatch hex={value} />;
  if (typeof value === 'boolean') return value ? 'Yes' : 'No';
  if (Array.isArray(value)) return `${value.length} item${value.length === 1 ? '' : 's'}`;
  return <code>{JSON.stringify(value)}</code>;
}

function ChangeDiff({ changes }: { changes: AuditLogEntry['changes'] }) {
  if (!changes) return null;
  // Detect shape
  const before = (changes as any).before as Record<string, unknown> | undefined;
  const after  = (changes as any).after  as Record<string, unknown> | undefined;

  // Flat-keyed extras (overwrite_target_*, etc.) — render before the diff.
  const flatKeys = Object.keys(changes).filter(k => k !== 'before' && k !== 'after');

  return (
    <div className="audit-diff">
      {flatKeys.map(key => (
        <div key={key} className="audit-diff__row">
          <span className="audit-diff__key">{key}:</span>{' '}
          {renderValue(key, (changes as any)[key])}
        </div>
      ))}
      {after && Object.keys(after).map(key => {
        const oldVal = before?.[key];
        const newVal = after[key];
        if (oldVal === newVal) return null;
        // Permission bitfield special case
        if ((key === 'permissions' || key === 'allow' || key === 'deny' || key === 'granted_permissions')
            && (typeof oldVal === 'number' || typeof oldVal === 'string' || oldVal === undefined)
            && (typeof newVal === 'number' || typeof newVal === 'string')) {
          const b = BigInt(oldVal ?? 0);
          const a = BigInt(newVal as any);
          return (
            <div key={key} className="audit-diff__row">
              <span className="audit-diff__key">{key}:</span>{' '}
              <PermDiff before={b} after={a} />
            </div>
          );
        }
        return (
          <div key={key} className="audit-diff__row">
            <span className="audit-diff__key">{key}:</span>{' '}
            <span className="audit-diff__before">{renderValue(key, oldVal)}</span>
            {' → '}
            <span className="audit-diff__after">{renderValue(key, newVal)}</span>
          </div>
        );
      })}
    </div>
  );
}
```

The PERM_NAMES list should mirror the bit constants in `frontend/src/lib/permissions.ts`. Read that file and keep them in sync.
